	\textbf{PREGUNTA 1.} (5 ptos) Sean $A$, $B$, $C$ y $D$ las matrices:
	\[
	A = \begin{pmatrix} 0 & 1 & 0 \\ 2 & 0 & 0 \\ 0 & 0 & -1 \end{pmatrix} \quad
	B = \begin{pmatrix} 4 & 3 & -3 \\ 6 & 1 & 9 \\ -3 & 0 & 5 \end{pmatrix} 
	\quad
	C = \begin{pmatrix} 0 & -1 & 0 \\ -2 & 0 & 0 \\ 0 & 0 & 2 \end{pmatrix} \quad
	D = \begin{pmatrix} 1 & 1 & -1 \\ 2 & 0 & 3 \\ -1 & 0 & 1 \end{pmatrix}
	\]
	Calcule (2.5 ptos c/u): $AC + 2I$ y $(B - 3D)^3$.

		\end{cases}
		\]
	\end{itemize}
